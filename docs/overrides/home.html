pipeline.py
from vectrix import Vectrix

vx = Vectrix(data)

# full pipeline
pipe = vx.pipeline([
  "log",
  "deseasonalize",
  "scale"
])
result = pipe.forecast(12)

# backtesting
bt = vx.backtest(
  steps=12,
  windows=5
)

# what-if scenarios
scenarios = vx.whatIf(
  shocks=[-0.2, 0, 0.2]
)
Advanced Features

Beyond basic forecasting

sklearn-Style Pipeline
